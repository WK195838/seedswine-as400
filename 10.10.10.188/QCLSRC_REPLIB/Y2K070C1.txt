                               (SC01B '%SST(SC01 1 1)' ))

                                         "' *CAT &COMPE *CAT '") +

                                         "' *CAT &DSC33E *CAT '") +
                        ') +

                        MAPFLD((MSG26 SG26 *CHAR 8) +

                          MAPFLD((MSN05 SN05 *CHAR 8) (SN01A +
