             PGM

             DCL        VAR(&JOB)   TYPE(*CHAR) LEN(10)
             DCL        VAR(&W#PROC) TYPE(*CHAR) LEN(1)

             DCL        VAR(&IN03)   TYPE(*LGL)  LEN(1)
             DCL        VAR(&IN10)   TYPE(*LGL)  LEN(1)
             DCL        VAR(&IN14)   TYPE(*LGL)  LEN(1)
             DCL        VAR(&IN96)   TYPE(*LGL)  LEN(1)

             RTVJOBA    JOB(&JOB)

             /* &W#PROC --> 0 =未處理, 1 =處理中*/
             RTVDTAARA  DTAARA(SCDA01 (3   1)) RTNVAR(&W#PROC)
             IF         COND(&W#PROC *EQ '1') THEN(DO)
             SNDBRKMSG  MSG('採購批次作業目前正在處理中，請稍待！+
                        ') TOMSGQ(&JOB)
             RETURN
             ENDDO

 START:      CALL       PGM(PUC0211) PARM(&IN03 &IN10 &IN14 &IN96)
             IF         COND(&IN03) THEN(DO)
             CHGVAR     VAR(%SST(*LDA 501 100)) VALUE(' ')
             RETURN
             ENDDO

             IF         COND(&IN14) THEN(DO)

             /* LOCK CHECK */

             ALCOBJ     OBJ((*LIBL/POCEPF *FILE *EXCL)) WAIT(0)
             MONMSG     MSGID(CPF1002) EXEC(DO)
             CHGVAR     VAR(&IN96) VALUE('1')
             GOTO       CMDLBL(RLSOBJ)
             ENDDO

             ALCOBJ     OBJ((*LIBL/POCHPF *FILE *EXCL)) WAIT(0)
             MONMSG     MSGID(CPF1002) EXEC(DO)
             CHGVAR     VAR(&IN96) VALUE('1')
             GOTO       CMDLBL(RLSOBJ)
             ENDDO

             ALCOBJ     OBJ((*LIBL/POCIPF *FILE *EXCL)) WAIT(0)
             MONMSG     MSGID(CPF1002) EXEC(DO)
             CHGVAR     VAR(&IN96) VALUE('1')
             GOTO       CMDLBL(RLSOBJ)
             ENDDO

             ALCOBJ     OBJ((*LIBL/POCKPF *FILE *EXCL)) WAIT(0)
             MONMSG     MSGID(CPF1002) EXEC(DO)
             CHGVAR     VAR(&IN96) VALUE('1')
             GOTO       CMDLBL(RLSOBJ)
             ENDDO

             ALCOBJ     OBJ((*LIBL/POCDPF *FILE *EXCL)) WAIT(0)
             MONMSG     MSGID(CPF1002) EXEC(DO)
             CHGVAR     VAR(&IN96) VALUE('1')
             GOTO       CMDLBL(RLSOBJ)
             ENDDO

             ALCOBJ     OBJ((*LIBL/POCJPF *FILE *EXCL)) WAIT(0)
             MONMSG     MSGID(CPF1002) EXEC(DO)
             CHGVAR     VAR(&IN96) VALUE('1')
             GOTO       CMDLBL(RLSOBJ)
             ENDDO

             ALCOBJ     OBJ((*LIBL/POCLPF *FILE *EXCL)) WAIT(0)
             MONMSG     MSGID(CPF1002) EXEC(DO)
             CHGVAR     VAR(&IN96) VALUE('1')
             GOTO       CMDLBL(RLSOBJ)
             ENDDO

             ALCOBJ     OBJ((*LIBL/POCMPF *FILE *EXCL)) WAIT(0)
             MONMSG     MSGID(CPF1002) EXEC(DO)
             CHGVAR     VAR(&IN96) VALUE('1')
             GOTO       CMDLBL(RLSOBJ)
             ENDDO

             ALCOBJ     OBJ((*LIBL/POCNPF *FILE *EXCL)) WAIT(0)
             MONMSG     MSGID(CPF1002) EXEC(DO)
             CHGVAR     VAR(&IN96) VALUE('1')
             GOTO       CMDLBL(RLSOBJ)
             ENDDO

             ALCOBJ     OBJ((*LIBL/POCOPF *FILE *EXCL)) WAIT(0)
             MONMSG     MSGID(CPF1002) EXEC(DO)
             CHGVAR     VAR(&IN96) VALUE('1')
             GOTO       CMDLBL(RLSOBJ)
             ENDDO

             /* 變更狀態碼 */
             CHGDTAARA  DTAARA(SCDA01 (3   1)) VALUE('1')

             CALL       PGM(PUC0211C)

             /* 執行完成,變更狀態 */
             CHGDTAARA  DTAARA(SCDA01 (3   1)) VALUE('0')

             ENDDO     /* IN14 */

             /* DEALLOCATE OBJECT    */
 RLSOBJ:     DLCOBJ     OBJ((*LIBL/POCEPF *FILE *EXCL))
             DLCOBJ     OBJ((*LIBL/POCHPF *FILE *EXCL))
             DLCOBJ     OBJ((*LIBL/POCIPF *FILE *EXCL))
             DLCOBJ     OBJ((*LIBL/POCKPF *FILE *EXCL))
             DLCOBJ     OBJ((*LIBL/POCDPF *FILE *EXCL))
             DLCOBJ     OBJ((*LIBL/POCJPF *FILE *EXCL))
             DLCOBJ     OBJ((*LIBL/POCLPF *FILE *EXCL))
             DLCOBJ     OBJ((*LIBL/POCMPF *FILE *EXCL))
             DLCOBJ     OBJ((*LIBL/POCNPF *FILE *EXCL))
             DLCOBJ     OBJ((*LIBL/POCOPF *FILE *EXCL))


             GOTO       CMDLBL(START)

             ENDPGM
